Map<String, dynamic> toJson() {

  data['thumbnailSecret'] = this.thumbnailSecret;
  data['height'] = this.height;
  data['width'] = this.width;

      EMFileMessageBody.downloadStatusToInt(this.thumbnailStatus);

}
